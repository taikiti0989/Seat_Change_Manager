# 席替え制約ファイル
# ==========================================
# 【書き方の種類】
#
# ▼ 仲が悪いペア（隣接させない）
#   bad_pair: 生徒番号A, 生徒番号B
#
# ▼ 視力が悪い（前から N 行以内に座らせる）
#   near_front: 生徒番号, 最大行数
#   ※ 行数は1始まり（1 = 最前列）
#
# ▼ 旧フォーマット互換（カンマ区切り2数字のみ）は「仲が悪いペア」として扱われます
#   例: 1,2  → 生徒①と②は隣接させない
#
# （# で始まる行はコメントとして無視されます）
# ==========================================

# --- 仲が悪いペアの例 ---
bad_pair: 1, 2
# 生徒③と⑤も離して座らせたい
bad_pair: 3, 5

# --- 視力が悪い生徒の例 ---
# 生徒④は目が悪いので、前から2行以内に座らせる
near_front: 4, 2
# 生徒⑦も視力が弱いので、前から3行以内に
near_front: 7, 3
